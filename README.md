long long calculate_even_indices_sum(const int matrix[ROWS][COLS]) {
    long long sum = 0;
    for (int i = 0; i < ROWS; ++i) {
        for (int j = 0; j < COLS; ++j) {
            if (i % 2 == 0 && j % 2 == 0) {
                sum += matrix[i][j];
            }
        }
    }
    return sum;
}

    }
}
